    static void Main(string[] args)
    {
        TaskService taskService = new TaskService();

        while (true)
        {
            ShowMenu();
            string choice = Console.ReadLine();

            try
            {
                switch (choice)
                {
                    case "1": Add(taskService); break;
                    case "2": ShowAll(taskService); break;
                    case "3": DeleteTask(taskService); break;
                    case "x":
                        Console.WriteLine("Exit the program....");
                        return;
                    default:
                        Console.WriteLine("Invalid choice!");
                        Pouse();
                        break;
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine(ex.Message);
                Pouse();
            }
        }
    }

    private static void DeleteTask(TaskService taskService)
    {
        Console.Clear();
        Console.WriteLine("\n<===== Deleting a task ======>");
        Console.WriteLine();
        var toDoList = taskService.GetAllTasks();
        for (int i = 0; i < toDoList.Count; i++)
        {
            Console.WriteLine($"{i + 1}.{toDoList[i]}");
        }
        Console.WriteLine();
        Console.Write("Enter number of task: ");
        int num = int.Parse(Console.ReadLine());
        taskService.RemoveTask(num - 1);
        Console.WriteLine();
        Console.WriteLine($"Task {num} is deleted.");
        Pouse();
    }

    private static void ShowAll(TaskService taskService)
    {
        Console.Clear();
        Console.WriteLine("<===== List of Tasks =====>");
        Console.WriteLine();
        var toDoList = taskService.GetAllTasks();
        for (int i = 0; i < toDoList.Count; i++)
        {
            Console.WriteLine($"{i + 1}.{toDoList[i]}");
        }
        Pouse();
    }

    private static void Add(TaskService taskService)
    {
        Console.Clear();
        Console.WriteLine("<===== Adding a task ======>");
        Console.WriteLine();
        Console.Write("Enter Task: ");
        string taskName = Console.ReadLine();
        taskService.AddTask(taskName);
        Console.WriteLine();
        Console.WriteLine("Task is added!");
        Pouse();
    }

    private static void ShowMenu()
    {
        Console.Clear();
        Console.WriteLine("<===== To - Do - List ======>");
        Console.WriteLine();
        Console.WriteLine("1.Add Task");
        Console.WriteLine("2.Show All Tasks");
        Console.WriteLine("3.Delete Task");
        Console.WriteLine("x.Exit");
        Console.WriteLine();
        Console.Write("Your choice is: ");
    }

    static void Pouse()
    {
        Console.WriteLine("\nPress ENTER to continue...");
        Console.ReadLine();
    }
}
